JSON chars

      

        " → \"

        \ → \\

        newline → \n

        tab → \t

        return → \r
      

    

    

      
XML chars

      

        & → &amp;

        < → &lt;

        > → &gt;

        " → &quot;

        ' → &apos;
      

    

    

      
HTML chars

      

        & → &amp;

        < → &lt;

        > → &gt;

        " → &quot;

        space → &nbsp;
      

    

    

      
Raw escapes

      

        \n → newline

        \r → return

        \t → tab

        \0 → null

        \\ → backslash
